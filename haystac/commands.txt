#Database creation
haystac database \
    --mode build \
    --output ../genomes_euka_ref/databaseforhaystac/ \
    --sequences-file seqfile_euka.txt  \


#Sample prep
haystac sample \
    --fastq euksim.combined.fq.gz \
    --output euka_haystac

#Analysis
haystac analyse \
    --mode abundances \
    --database ../reference/databaseforhaystac/ \
    --sample euka_haystac \
    --output euka_haystac_abundances \
    --mapdamage True \
